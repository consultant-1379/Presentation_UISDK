/**
 * Allows low level access to Titan in order to create extensions.
 * See [Confluence](http://confluence-oss.lmera.ericsson.se/display/TORFTUI/Client+framework+API+Documentation#ClientframeworkAPIDocumentation-extendtitan) for instructions on how to create an extension.
 *
 * @method extension
 * @param fn {Function} An anonymous function that wraps extension logic. The first param in this function will give
 *                      access to $, _, Backbone, Handlebars, and _private. This function also has direct access to Titan.
 */
Titan.extension = function (fn) {
    fn.call(this, {
        $: $,
        _: _,
        Backbone: Backbone,
        Handlebars: Handlebars,
        _private: _private
    });
};

/**
 * Creates a new instance of a specified object with a reference to the specified parent, and adds itself as a child of
 * the specified parent.
 *
 * @method create
 * @param MyClass {Object} The object we want to create an instance of
 * @param parent {*} The parent that this object should be a child of
 * @return {*} The initialised object
 */
Titan.create = function(MyClass, parent) {
    var child = new MyClass(parent, Titan.utils.getListFromArguments(arguments, 2));
    parent.addChild(child);
    return child;
};

/**
 * Traverses up the parent/child tree until it gets to the root Application.
 *
 * @method Utils_getApplication
 * @return {Titan.Application} The root Application
 */
function Utils_getApplication() {
    var self = this;
    while(self.parent !== undefined) {

        self = self.parent();
        if(self === undefined){
            return;
        }

    }

    return self;
}

/**
 * Applies special properties directly to the callee.
 *
 * @private
 * @method Utils_configure
 * @param options {Object} A JavaScript object with properties to be added to the callee's options object
 * @param names {Array} The names of properties to be applied directly to the callee from the options object
 */
_private.Utils_configure = Utils_configure;
function Utils_configure(options, names) {
    if (this.options) {
        options = _.extend({}, this.options, options);
    }
    names.forEach(function (attr) {
        if (options[attr]) {
            this[attr] = options[attr];
        }
    }, this);
    this.options = options;
}

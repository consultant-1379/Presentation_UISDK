/**
 * Connects the Model to the View
 *
 * @class Titan.Presenter
 * @constructor
 * @param options {Object} Contains optional properties that are added to the Presenter's options object.
 *                         Applies special options (model, collection, eventBus, view) directly to the Presenter.
 */
var Presenter = Titan.Presenter = function (parent, options) {

    Utils_configure.call(this, options || {}, ['model', 'collection']);

    this._children = [];
    this._parent = parent;
    this.init(Utils_getApplication.call(this));

};

/**
 * Provides access to Backbone.View.extend via _private.Utils_extend
 * See: [http://backbonejs.org/#View-extend](http://backbonejs.org/#View-extend)
 *
 * @method extend
 */
Presenter.extend = Utils_extend;

Presenter.prototype = {

    /**
     * Method called when Presenter is initialised. Empty by default, it can be overridden to provide functionality
     *
     * @method init
     */
    init: function (app) {
    },

    /**
     * Instantiates and returns a given class. The class is given a reference to the Presenter as a parent, and is
     * added as a child of Presenter.
     *
     * @param MyClass {*} The Class to be instantiated
     * @return {*} An instance of MyClass
     */
    create: function(MyClass) {
        return Titan.create(MyClass, this);
    },

    /**
     * Replaces the contents of the specified container with the current widget
     *
     * @method start
     * @param container {Titan.Element} The Element to place the widget within
     */
    start: function (container) {
        if (this.view) {
            var parent = View_getRoot.call(this.view).parent();
            if (parent.length === 0 || parent[0] !== Element_getRoot.call(container)[0]) {
                container.setWidget(this.view);
            }
        }
    },
    
    /**
     * Removes the View associated with the Presenter from the container
     *
     * @param container {Titan.Element} The container that the application is currently running in.
     */
    stop: function (container) {
        this.beforeStop();

/**
 * Encapsulates the entire Application. Eliminates the need to manually create an Element, Router and
 * MainController, and begin tracking of the hash value on the router. Triggers the initialisation of the whole app.
 *
 * @class Titan.Application
 * @constructor
 * @param options {Object} Contains optional properties that are added to the Application's options object.
 *                         Applies special options (eventBus, container) directly to the Application.
 */
var Application = Titan.Application = function (options) {
    if (!options) {
        options = {};
    }

    this._children = [];
    this.globalEvents = options.eventBus || new Titan.EventBus();
    this.localEvents = new Titan.EventBus();
    if (options.dictionary) {
        this.dictionary = options.dictionary;
    } else {
        this.dictionary = this.dictionary || {};
    }
    if (options.namespace) {
        this.namespace = options.namespace;
    } else {
        this.namespace = this.namespace || {};
    }

    if (options.container instanceof Titan.Element) {
        this.container = options.container;
    } else {
        var $el = $(options.container);
        this.container = new Titan.Element($el);
    }
};

/**
 * Provides access to Backbone.View.extend via _private.Utils_extend
 * See: [http://backbonejs.org/#View-extend](http://backbonejs.org/#View-extend)
 *
 * @method extend
 */
Application.extend = Utils_extend;

Application.prototype = {

    /**
     * Adds a specified object as a child of the Application.
     *
     * @method addChild
     * @param child {Object} The object to be added as a child
     */
    addChild: function (child) {
        this._children.push(child);
    },

    /**
     * Creates a new Router and MainController with the EventBus, places and container for the application, and
     * begins tracking changes to the route.
     *
     * @method start
     * @param options {Object} Contains optional properties to be used on start.
     */
    start: function (options) {
        /*
         var router,
         mainController;*/

        this.router = new Titan.Router({
            eventBus: this.localEvents,
            places: this.places,
            silent: options && options.silent
        });

        //noinspection JSUnusedAssignment
        this.mainController = new Titan.MainController({
            container: this.container,
            application: this
        });
        this.router.handleCurrentState();
    },

    /**
     * Stops the main controller and deletes it from memory.
     *
     * @method stop
     */
    stop: function () {
        if (this.mainController !== undefined) {
            _.each(this._children, function (child) {
                child.stop();
            });

            this.mainController.destroy();
            delete this.mainController;
        }
    }

};
